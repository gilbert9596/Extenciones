```
	"prefix": "clg",
	"body": [
		"console.log(${1:object});", 
		"$2"
	],
	"description": "Mostrar un log en la consola..."

},
```

```
	"prefix": "clase",
	"body": [
		"class ${1:NombreClase} {",

		"   }",	
		"}"

}
```
